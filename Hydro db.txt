CREATE TABLE hyd_detalle_ingreso (
 DET_DETALLE_INGRESO int(11) NOT NULL AUTO_INCREMENT,
 IMP_INGRESO_MARTERIA_PRIMA int(11) DEFAULT NULL,
 PRD_PRODUCTO int(11) DEFAULT NULL,
 DET_CANTIDAD int(11) DEFAULT NULL,
 DET_PESONETO int(11) DEFAULT NULL,
 DET_PESOTARA int(11) DEFAULT NULL,
 DET_DENSIDAD varchar(50) DEFAULT NULL,
 DET_HUMEDAD int(11) DEFAULT NULL,
 MED_MEDIDA int(11) DEFAULT NULL,
 PRIMARY KEY (DET_DETALLE_INGRESO)
); 

CREATE TABLE hyd_ingeso_materia_prima (
 IMP_INGRESO_MATERIA_PRIMA int(11) NOT NULL AUTO_INCREMENT,
 PRO_PROVEEDOR int(11) DEFAULT NULL,
 IMP_FECHA date DEFAULT NULL,
 IMP_LOTE int(11) DEFAULT NULL,
 IMP_PLACASVEHICULO varchar(20) DEFAULT NULL,
 USU_USUARIO varchar(20) DEFAULT NULL,
 PRIMARY KEY (IMP_INGRESO_MATERIA_PRIMA)
);

CREATE TABLE hyd_medida (
 MED_MEDIDA int(11) NOT NULL AUTO_INCREMENT,
 MED_DESCRIPCION varchar(50) DEFAULT NULL,
 PRIMARY KEY (MED_MEDIDA)
); 

CREATE TABLE hyd_persona (
 PER_PERSONA int(11) NOT NULL AUTO_INCREMENT,
 PER_PRIMERNOMBRE varchar(30) DEFAULT NULL,
 PER_SEGUNDONOMBRE varchar(30) DEFAULT NULL,
 PER_PRIMERAPELLIDO varchar(30) DEFAULT NULL,
 PER_SEGUNDOAPELLIDO varchar(30) DEFAULT NULL,
 PER_DPI varchar(20) DEFAULT NULL,
 PER_DIRECCION varchar(80) DEFAULT NULL,
 PER_TELEFONO int(11) DEFAULT NULL,
 PER_EMAIL varchar(50) DEFAULT NULL,
 PUE_PUESTO int(11) DEFAULT NULL,
 PRIMARY KEY (PER_PERSONA)
);

CREATE TABLE hyd_producto (
 PRD_PRODUCTO int(11) NOT NULL AUTO_INCREMENT,
 PRD_DESCRIPCION varchar(50) DEFAULT NULL,
 PRIMARY KEY (PRD_PRODUCTO)
) ;

CREATE TABLE hyd_proveedor (
 PRO_PROVEEDOR int(11) NOT NULL AUTO_INCREMENT,
 PRO_NOMBRE varchar(50) DEFAULT NULL,
 PRO_REGION varchar(50) DEFAULT NULL,
 PRO_LOTE int(11) DEFAULT NULL,
 PRO_DIRECCION varchar(80) DEFAULT NULL,
 PRO_NIT varchar(20) DEFAULT NULL,
 PRO_TELEFONO int(11) DEFAULT NULL,
 PRIMARY KEY (PRO_PROVEEDOR)
);

CREATE TABLE hyd_puesto (
 PUE_PUESTO int(11) NOT NULL AUTO_INCREMENT,
 PUE_DESCRIPCION varchar(50) DEFAULT NULL,
 PRIMARY KEY (PUE_PUESTO)
) ;

CREATE TABLE hyd_rol (
 Rol_rol int(11) NOT NULL AUTO_INCREMENT,
 ROL_Descripcion varchar(50) DEFAULT NULL,
 PRIMARY KEY (Rol_rol)
) ;

CREATE TABLE hyd_usuario (
 USU_Usuario int(11) NOT NULL AUTO_INCREMENT,
 USU_Nombre varchar(50) DEFAULT NULL,
 USU_Contrasenia varchar(20) DEFAULT NULL,
 PER_Persona int(11) DEFAULT NULL,
 ROL_Rol int(11) DEFAULT NULL,
 PRIMARY KEY (USU_Usuario)
) ;
